145 inline bool next_pos_up()
146 {
147 ++k;
148 if (partdist[k] != 0.0)
149 {
150 x[k] += dx[k];
151 ddx[k] = -ddx[k];
152 dx[k] = ddx[k] - dx[k];
153 }
154 else
155 {
156 if (k >= k_end)
157 return false;
158 k_max = k;
159 if (is_svp)
160 {
161 // only for SVP: break symmetry at the top non-zero coefficient
162 ++x[k];
163 }
164 else
165 {
166 x[k] += dx[k];
167 ddx[k] = -ddx[k];
168 dx[k] = ddx[k] - dx[k];
169 }
170 }
171 return true;
172 }
