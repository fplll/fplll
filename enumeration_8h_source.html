155  inline void enumerate_recur(i_tag<i, svp, swirl, swirlid>)
156  {

158  _r[i - 1] = _r[i];

160  float_type yi = round(ci);
161  int xi = (int)(yi);
162  yi = ci - yi;
163  float_type li = _l[i + 1] + (yi * yi * risq[i]);
164 #ifndef NOCOUNTS
165  ++_counts[i];
166 #endif
167 
168  if (findsubsols && li < _subsolL[i] && li != 0.0)
169  {
170  _subsolL[i] = li;
171  _subsol[i][i] = xi;
172  for (int j = i + 1; j < N; ++j)
173  _subsol[i][j] = _x[j];
174  }
175  if (li > _AA[i])
176  return;
177 
178  _Dx[i] = _D2x[i] = (((int)(yi >= 0) & 1) << 1) - 1;
179  _c[i] = ci;
180  _x[i] = xi;
181  _l[i] = li;
182 
183  for (int j = _r[i - 1]; j > i - 1; --j)
184  _sigT[i - 1][j - 1] = _sigT[i - 1][j] - _x[j] * muT[i - 1][j];

186 #ifndef NOLOCALUPDATE

188 #endif

192  enumerate_recur(i_tag<i - 1, svp, swirl, swirlid>());
193

195  {
196  ++_x[i];
197  _r[i - 1] = i;
198  float_type yi2 = _c[i] - _x[i];
199  float_type li2 = _l[i + 1] + (yi2 * yi2 * risq[i]);
200  if (li2 > _AA2[i])
201  return;
202  _l[i] = li2;
203  _sigT[i - 1][i - 1] = _sigT[i - 1][i] - _x[i] * muT[i - 1][i];

208  _D2x[i] = -_D2x[i];
209  _Dx[i] = _D2x[i] - _Dx[i];
210  _r[i - 1] = i;
211  float_type yi2 = _c[i] - _x[i];

382  template <bool svp = true> void enumerate_recursive()
383  {
384  for (size_t i = 0; i < globals.signal.size(); ++i)
385  globals.signal[i] = 0;
386  threadid = get_threads();
387 
388  _A = globals.A;
389  _update_AA();
390 
391  for (int j = 0; j < N; ++j)
392  {
393  _x[j] = _Dx[j] = 0;
394  _D2x[j] = -1;
395  _sol[j] = 0;
396  _c[j] = _l[j] = 0.0;
397  _subsolL[j] = risq[j];
398  for (int k = 0; k < N; ++k)
399  {
400  _sigT[j][k] = 0.0;
401  _subsol[j][k] = 0;
402  }
403  _r[j] = N - 1;
404  _counts[j] = 0;
405  }
406  _l[N] = 0.0;
407  _counts[N] = 0;
408 
409 #ifdef SINGLE_THREADED
410  enumerate_recur(i_tag<N - 1, svp, -2, 0>());
411 #else
412  auto &swirlys = globals.swirlys;
413  swirlys.resize(2);
414  swirlys[0].clear();
415  enumerate_recur(i_tag<N - 1, svp, N - SWIRLY, 0>());
416 
417  const auto swirl_less = [](const swirl_item_t &l, const swirl_item_t &r) {
418  return l.second.second < r.second.second;
419  };
420  if (activeswirly)
421  {
422  sort(swirlys[0].begin(), swirlys[0].end(), swirl_less);
423  }
424 
425  size_t swirly0idx = 0;
426  swirlys[1].clear();
427  while (swirly0idx < swirlys[0].size())
428  {
429  int swirly1newstart = (int)(swirlys[1].size());
430  while (swirly0idx < swirlys[0].size() && swirlys[1].size() < SWIRLY2BUF)
431  {
432  const int i = N - SWIRLY;
433  _x = swirlys[0][swirly0idx].first;
434  _l[i] = swirlys[0][swirly0idx].second.first;
435  for (int j = 0; j < N; ++j)
436  _r[j] = N - 1;
437  for (int j = N - 1; j > i - 1; --j)
438  _sigT[i - 1][j - 1] = _sigT[i - 1][j] - _x[j] * muT[i - 1][j];
439 
440  enumerate_recur(i_tag<N - SWIRLY - 1, svp, N - 2 * SWIRLY, 1>());
441 
442  ++swirly0idx;
443  }
444 
445  size_t swirly1end = (int)(swirlys[1].size());
446  if (activeswirly)
447  {
448  // sort the new additions to swirly1
449  sort(swirlys[1].begin() + swirly1newstart, swirlys[1].end(), swirl_less);
450  // merge with previous elms in swirly1
451  inplace_merge(swirlys[1].begin(), swirlys[1].begin() + swirly1newstart, swirlys[1].end(),
452  swirl_less);
453 
454  // process portion of swirly[1] and then add more
455  swirly1end = (SWIRLY2BUF >> SWIRLY1FRACTION);
456  if (swirly1end > swirlys[1].size())
457  swirly1end = swirlys[1].size();
458  }
459 
460  auto &swirly_ref = swirlys[1];
461  std::atomic<std::size_t> swirly_i(0);
462  unsigned threadid = 0;
463  auto f = [this, &swirly_ref, &swirly_i, swirly1end, &threadid]() {
464  auto mylat = *this;
465  {
466  lock_type lock(globals.mutex);
467  mylat.threadid = threadid++;
468  }
469  for (int j = 0; j < N - SWIRLY; ++j)
470  mylat._counts[j] = 0;
471  while (true)
472  {
473  std::size_t idx = swirly_i++;
474  if (idx >= swirly1end)
475  break;
476 
477  const int i = N - 2 * SWIRLY;
478  mylat._x = swirly_ref[idx].first;
479  mylat._l[i] = swirly_ref[idx].second.first;
480  for (int j = 0; j < N; ++j)
481  mylat._r[j] = N - 1;
482  for (int j = N - 1; j > i - 1; --j)
483  mylat._sigT[i - 1][j - 1] = mylat._sigT[i - 1][j] - mylat._x[j] * mylat.muT[i - 1][j];
484 
485  mylat._thread_local_update();
486 
487  mylat.enumerate_recur(i_tag<N - 2 * SWIRLY - 1, svp, -2, -1>());
488  }
489 
490  lock_type lock(globals.mutex);
491  for (int j = 0; j < N - SWIRLY; ++j)
492  this->_counts[j] += mylat._counts[j];
493  for (int j = 0; j < N; ++j)
494  if (mylat._subsolL[j] < this->_subsolL[j])
495  {
496  this->_subsolL[j] = mylat._subsolL[j];
497  this->_subsol[j] = mylat._subsol[j];
498  }
499  };
500  for (int i = 0; i < ::fplll::get_threads(); ++i)
501  threadpool.push(f);
502  threadpool.wait_work();
503 
504  swirlys[1].erase(swirlys[1].begin(), swirlys[1].begin() + swirly1end);
505  }
506 #ifndef NOCOUNTS
507 // if (enumlib_loglevel >= 1) cout << "[enumlib] counts: " << _counts << endl;
508 #endif
509 #endif
510  }
