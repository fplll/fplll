41 template <class T> void rotate_by_swap(vector<T> &v, int first, int middle, int last)
42 {
43  // Algorithm from STL code
44  reverse_by_swap(v, first, middle - 1);
45  reverse_by_swap(v, middle, last);
46  for (; first < middle && middle <= last; first++, last--)
47  {
48  v[first].swap(v[last]);
49  }
50  if (first == middle)
51  reverse_by_swap(v, middle, last);
52  else
53  reverse_by_swap(v, first, middle - 1);
54 }

92 template <class T> istream &operator>>(istream &is, vector<T> &v)
93 {
94  char c;
95  v.clear();
96  if (!(is >> c))
97  return is;
98  if (c != '[')
99  {
100  is.setstate(ios::failbit);
101  return is;
102  }
103  while (is >> c && c != ']')
104  {
105  is.putback(c);
106  v.resize(v.size() + 1);
107  if (!(is >> v.back()))
108  {
109  v.pop_back();
110  return is;
111  }
112  }
113  return is;
114 }
