148 template <class ZT, class FT> inline void HLLLReduction<ZT, FT>::compute_dR(int k)
149 {
150  m.get_R(dR[k], k, k);
151  dR[k].mul(dR[k], dR[k]);
152  dR[k].mul(delta, dR[k]); // dR[k] = delta * R(k, k)^2
153 }
