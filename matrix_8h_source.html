84 void dot_product(T &result, const MatrixRow<T> &v1, const MatrixRow<T> &v2, int n)
85 {
86  FPLLL_DEBUG_CHECK(n > 0 && n <= v1.size() && v1.size() == v2.size() &&
87  (v1.is_zero(n) || v2.is_zero(n)));
88  result.mul(v1[0], v2[0]);
89  for (int i = 1; i < n; i++)
90  {
91  result.addmul(v1[i], v2[i]);
92  }
93 }
