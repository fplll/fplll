106 template <class ZT, class FT>
107 inline bool LLLReduction<ZT, FT>::size_reduction(int kappa_min, int kappa_end,
108  int size_reduction_start)
109 {
110  if (kappa_end == -1)
111  kappa_end = m.d;
112  for (int k = kappa_min; k < kappa_end; k++)
113  {
114  if ((k > 0 && !babai(k, k, size_reduction_start)) || !m.update_gso_row(k))
115  return false;
116  }
117  return set_status(RED_SUCCESS);
118 }

120 template <class ZT, class FT>
121 inline bool LLLReduction<ZT, FT>::early_reduction(int start, int size_reduction_start)
122 {
123  m.lock_cols();
124  if (verbose)
125  {
126  cerr << "Early reduction start=" << start + 1 << endl;
127  }
128  for (int i = start; i < m.d; i++)
129  {
130  if (!babai(i, start, size_reduction_start))
131  return false;
132  }
133  m.unlock_cols();
134  last_early_red = start;
135  return true;
136 }
