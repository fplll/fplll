106 template <class ZT, class FT>
107 inline bool LLLReduction<ZT, FT>::size_reduction(int kappa_min, int kappa_end,
108  int size_reduction_start)
109 {
110  if (kappa_end == -1)
111  kappa_end = m.d;
112 
113  extend_vect(babai_mu, kappa_end);
114  extend_vect(babai_expo, kappa_end);
115 
116  for (int k = kappa_min; k < kappa_end; k++)
117  {
118  if ((k > 0 && !babai(k, k, size_reduction_start)) || !m.update_gso_row(k))
119  return false;
120  }
121  return set_status(RED_SUCCESS);
122 }

124 template <class ZT, class FT>
125 inline bool LLLReduction<ZT, FT>::early_reduction(int start, int size_reduction_start)
126 {
127  m.lock_cols();
128  if (verbose)
129  {
130  cerr << "Early reduction start=" << start + 1 << endl;
131  }
132  for (int i = start; i < m.d; i++)
133  {
134  if (!babai(i, start, size_reduction_start))
135  return false;
136  }
137  m.unlock_cols();
138  last_early_red = start;
139  return true;
140 }
