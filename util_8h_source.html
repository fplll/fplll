62 template <class T>
63 void scalar_product(T &result, const MatrixRow<T> &v1, const MatrixRow<T> &v2, int n)
64 {
65  FPLLL_DEBUG_CHECK(n <= static_cast<int>(v1.size()) && n <= static_cast<int>(v2.size()));
66  T tmp;
67  result.mul(v1[0], v2[0]);
68  for (int i = 1; i < n; i++)
69  {
70  tmp.mul(v1[i], v2[i]);
71  result.add(result, tmp);
72  }
73 }
