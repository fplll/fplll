258  inline void thread_pool::resize(std::size_t nrthreads)
259  {
260  if (nrthreads < _threads.size())
261  {
262  // decreasing number of active threads
263  std::unique_lock<std::mutex> lock(_mutex);
264  for (std::size_t i = nrthreads; i < _threads.size(); ++i)
265  *(_threads_stop[i]) = true;
266  _condition.notify_all();
267  lock.unlock();
268  for (std::size_t i = nrthreads; i < _threads.size(); ++i)
269  _threads[i]->join();
270 
271  lock.lock();
272  _threads_stop.resize(nrthreads);
273  _threads.resize(nrthreads);
274  }
275  else if (nrthreads > _threads.size())
276  {
277  // wait before resizing because it may cause reallocation
278  wait_work();
279 
280  std::unique_lock<std::mutex> _lock(_mutex);
281  _threads_stop.reserve(nrthreads);
282  _threads.reserve(nrthreads);
283  for (std::size_t i = _threads.size(); i < nrthreads; ++i)
284  {
285  _threads_stop.emplace_back(new std::atomic_bool(false));

287  }

291  inline void thread_pool::_init_thread()
292  {
293  std::size_t i = _threads.size();
294  if (i >= _threads_stop.size())
295  throw std::runtime_error("thread_pool::_init_thread(): index out of range!");
296  auto f = [this,i]() {
297  std::function<void()> task;
298  std::unique_lock<std::mutex> lock(this->_mutex);
299  while (true) {
300  if (this->_tasks.empty())
301  {
302  if (*(this->_threads_stop[i]))
303  return;
304  this->_condition.wait(lock);
305  continue;
306  }
307 
308  ++this->_threads_busy;
309  task = std::move(this->_tasks.front());
310  this->_tasks.pop();
311 
312  lock.unlock();
313  task();
314  --this->_threads_busy;
315  lock.lock();
316  }
317  };
318  _threads.emplace_back(new std::thread(f));
319  }
