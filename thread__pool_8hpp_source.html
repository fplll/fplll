173 {

176 std::lock_guard<std::mutex> lock(_mutex);

219 inline auto thread_pool::enqueue(F&& f, Args&&... args)
220 -> std::future<typename std::result_of<F(Args...)>::type>
221 {
222 typedef typename std::result_of<F(Args...)>::type return_type;
223 auto task = std::make_shared< std::packaged_task<return_type()> >
224 ( std::bind(std::forward<F>(f), std::forward<Args>(args)...) );
225 push( [task](){ (*task)(); } );
226 return task->get_future();

271 inline void thread_pool::resize(std::size_t nrthreads)
272 {
273 if (nrthreads < _threads.size())
274 {
275 // decreasing number of active threads
276 std::unique_lock<std::mutex> lock(_mutex);
277 for (std::size_t i = nrthreads; i < _threads.size(); ++i)
278 *(_threads_stop[i]) = true;
279 _condition.notify_all();
280 lock.unlock();
281 for (std::size_t i = nrthreads; i < _threads.size(); ++i)
282 _threads[i]->join();
283
284 lock.lock();
285 _threads_stop.resize(nrthreads);
286 _threads.resize(nrthreads);
287 }
288 else if (nrthreads > _threads.size())
289 {
290 // wait before resizing because it may cause reallocation
291 wait_work();
292
293 std::unique_lock<std::mutex> _lock(_mutex);
294 _threads_stop.reserve(nrthreads);
295 _threads.reserve(nrthreads);
296 for (std::size_t i = _threads.size(); i < nrthreads; ++i)
297 {
298 _threads_stop.emplace_back(new std::atomic_bool(false));
299 _init_thread();
300 }
301 }

304 inline void thread_pool::_init_thread()
305 {
306 std::size_t i = _threads.size();
307 if (i >= _threads_stop.size())
308 throw std::runtime_error("thread_pool::_init_thread(): index out of range!");
309 auto f = [this,i]() {
310 std::function<void()> task;
311 std::unique_lock<std::mutex> lock(this->_mutex);
312 while (true) {
313 if (this->_tasks.empty())
314 {
315 if (*(this->_threads_stop[i]))
316 return;
317 this->_condition.wait(lock);
318 continue;
319 }
320
321 ++this->_threads_busy;
322 task = std::move(this->_tasks.front());
323 this->_tasks.pop();
324
325 lock.unlock();
326 task();
327 --this->_threads_busy;
328 lock.lock();
329 }
330 };
331 _threads.emplace_back(new std::thread(f));
332 }

348 inline void barrier::wait()
349 {
350 std::unique_lock<std::mutex> lock(_mutex);
351 if (++_i >= _count)
352 {
353 _i = 0;
354 lock.unlock();
355 _condition.notify_all();
356 }
357 else
358 {
359 _condition.wait(lock);
360 }
